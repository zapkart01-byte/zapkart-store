// Backend: POST /auth/send-otp
// Called by mobile apps to trigger OTP
async function sendOTP(phone, userType) {
  const url = `https://2factor.in/API/V1/${process.env.TWOFACTOR_API_KEY}/SMS/${phone}/AUTOGEN`
  const response = await fetch(url)
  const { Details: sessionId } = await response.json()
  // Store sessionId in Redis with 120 second TTL
  await redis.setex(`otp:${phone}`, 120, sessionId)
  return { success: true }
}

// Backend: POST /auth/verify-otp
async function verifyOTP(phone, otp) {
  const sessionId = await redis.get(`otp:${phone}`)
  if (!sessionId) return { error: 'OTP expired' }
  const url = `https://2factor.in/API/V1/${process.env.TWOFACTOR_API_KEY}/SMS/VERIFY/${sessionId}/${otp}`
  const response = await fetch(url)
  const { Details: status } = await response.json()
  if (status !== 'OTP Matched') return { error: 'Invalid OTP' }
  // Find or create user in Supabase
  const { data: user } = await supabase.from(userType).select('*').eq('phone', phone).single()
  const token = generateJWT(user) // sign with JWT_SECRET
  return { token, user }
}

4.2 Platform Markup â€” INR 1 Per Item
How Platform Markup Works
Store lists milk at INR 29 (their selling price).
Platform adds INR 1 markup automatically.
Customer sees INR 30 on product card.
Customer pays INR 30. Store receives payout on INR 29.
ZapKart keeps the INR 1 markup in full.
Rule: customer_price = minimum of (store_price + 1) and (platform_mrp).
Never show store_price to customer. Always show customer_price.
Never mention markup anywhere in the customer app.

4.4 Dynamic Delivery Fee Formula
// pricingCalculator.js â€” IDENTICAL in admin, customer app, and backend
function calculateOrderPricing(cartItems, distanceKm, settings) {

  // Step 1: Rider base payout
  let riderPayout
  if (distanceKm < 2)      riderPayout = settings.rider_payout_under_2km  // 25
  else if (distanceKm < 4) riderPayout = settings.rider_payout_2_to_4km   // 35
  else                     riderPayout = settings.rider_payout_above_4km   // 50

  // Step 2: Commission from each item using its category rate
  const totalCommission = cartItems.reduce((sum, item) =>
    sum + (item.store_price * item.quantity * item.category_commission_rate), 0)

  // Step 3: Markup revenue (INR 1 per item, capped at MRP)
  const markupRevenue = cartItems.reduce((sum, item) => {
    const cp = Math.min(item.store_price + settings.platform_markup_per_item, item.platform_mrp)
    return sum + ((cp - item.store_price) * item.quantity)
  }, 0)

  // Step 4: Cart value from customer perspective
  const cartValue = cartItems.reduce((sum, item) => {
    const cp = Math.min(item.store_price + settings.platform_markup_per_item, item.platform_mrp)
    return sum + (cp * item.quantity)
  }, 0)

  // Step 5: Tiered minimum profit
  const minimumProfit = getMinimumProfit(cartValue, settings)

  // Step 6: Delivery fee to fill the gap
  let deliveryFee = (riderPayout + minimumProfit) - totalCommission - markupRevenue
  if (cartValue >= settings.free_delivery_above) deliveryFee = 0
  deliveryFee = Math.max(deliveryFee, settings.min_delivery_fee)
  deliveryFee = Math.min(deliveryFee, settings.max_delivery_fee)
  deliveryFee = Math.round(deliveryFee)

  return {
    cartValue, deliveryFee, riderPayout,
    totalCommission:   Math.round(totalCommission),
    markupRevenue:     Math.round(markupRevenue),
    zapkartNetProfit:  Math.round(totalCommission + markupRevenue + deliveryFee - riderPayout),
    isFreeDelivery:    cartValue >= settings.free_delivery_above,
    totalCustomerPays: cartValue + deliveryFee,
  }
}

function getMinimumProfit(cartValue, settings) {
  if (cartValue <= 149) return settings.min_profit_tier1  // INR 12
  if (cartValue <= 249) return settings.min_profit_tier2  // INR 14
  if (cartValue <= 399) return settings.min_profit_tier3  // INR 15
  if (cartValue <= 499) return settings.min_profit_tier4  // INR 10
  return settings.min_profit_tier5                         // INR 8
}
